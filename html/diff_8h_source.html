215  void compute_trace(std::vector<T> original, std::vector<T> updated) {
216  int n = original.size();
217  int m = updated.size();
218  int max = n + m;
219 
220  // Compute the trace.
221  std::vector<int> v(2 * max + 1);
222  v[1+max] = 0;
223  int d, k, x, y;
224  for (d = 0; d <= max; d=d+1) {
225  for (k = -d; k <= d; k=k+2) {
226  if ( k == -d || (k != d && v[k - 1 + max] < v[k + 1 + max])) {
227  x = v[k + 1 + max];
228  }
229  else {
230  x = v[k - 1 + max] + 1;
231  }
232  y = x - k;
233  while (x < n && y < m && original[x] == updated[y]) {
234  // Skip over common substring
235  x = x + 1;
236  y = y + 1;
237  }
238  v[k + max] = x;
239  if (x >= n && y >= m) {
